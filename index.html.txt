<script>
  // GANTI dengan URL Web App Google Apps Script
  const GOOGLE_SCRIPT_URL = "https://script.google.com/macros/s/AKfycbzkYQ_6tymzySmeoIaEP1WPWTwEkLGBR3uFnVFCa3n49XkJwzUnTK7PAFGyWo4rIHeELQ/exec";

  // Mapping ID Timbangan → Nama
  const dataTimbangan = {
    "TB-012": "Timbangan Hopper 1",
    "TB-013": "Timbangan Hopper 2",
    "TB-014": "Timbangan Packing"
  };

  // Ambil parameter ID dari URL
  const params = new URLSearchParams(window.location.search);
  const idTimbangan = params.get("id");

  document.getElementById("id_timbangan").value = idTimbangan || "Tidak Diketahui";
  document.getElementById("nama_timbangan").value =
    dataTimbangan[idTimbangan] || "Belum Terdaftar";

  function kirimData() {
    const data = {
      id_timbangan: document.getElementById("id_timbangan").value,
      nama_timbangan: document.getElementById("nama_timbangan").value,
      kegiatan: document.getElementById("kegiatan").value,
      sebelum_beban: document.getElementById("sebelum").value,
      setelah_beban: document.getElementById("setelah").value,
      beban_standar: document.getElementById("beban").value,
      petugas: document.getElementById("petugas").value
    };

    document.getElementById("status").innerHTML = "⏳ Mengirim data...";

    fetch(GOOGLE_SCRIPT_URL, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(data)
    })
    .then(res => res.json())
    .then(res => {
      if (res.success) {
        document.getElementById("status").innerHTML = "✅ Data berhasil disimpan";
        document.getElementById("sebelum").value = "";
        document.getElementById("setelah").value = "";
      } else {
        document.getElementById("status").innerHTML = "❌ Gagal menyimpan data";
      }
    })
    .catch(() => {
      document.getElementById("status").innerHTML = "❌ Koneksi bermasalah";
    });
  }
</script>
